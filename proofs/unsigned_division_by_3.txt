==== Unsigned division by 3

.Implementation
[source]
----
inline unsigned udiv3(uint32_t dividend, unsigned *remainder) {
    uint32_t q = (dividend * UINT64_C(0xAAAAAAAB)) >> 33;
    if(remainder) *remainder = dividend - q * 3;
    return q;
}
----
